 17 var base   = require('./base.js')
 18   , Page   = require('./page.js')
 19 
 20 /**
 21   @name Site
 22   @constructor
 23   @description Constructs a new instance of Site
 24   @param {object} config A site configuration object
 25 */
 26 function Site(controller, config){
 27   /**
 28     @name Site#controller
 29     @public
 30     @type BaseController
 31     @description The main controller for the application. 
 32     Needed for internal functions
 33   */
 34   this.controller = controller || {};
 35 
 36   /**
 37     @name Site#cdn
 38     @public
 39     @type Object
 40     @description The CDN configuration
 41   */
 42   this.cdn = config.cdn || {};
 43 
 44    /**
 45     @name Site#app
 46     @public
 47     @type SinglePageApplication
 48     @description The main application
 49   */
 50   this.app = this.controller.spa || {};
 51 
 52   /**
 53     @name Site#scope
 54     @public
 55     @type Object
 56     @description The current scope for the site, a layout or template.
 57   */
 58   this.scope = {};
 59 
 60   /**
 61     @name Site#bindings
 62     @public
 63     @type Object
 64     @description The variable and function bindings inherited from the scope 
 65   */
 66   this.bindings = {};
 67 
 68   /**
 69     @name Site#config
 70     @public
 71     @type Object
 72     @description The site configuration object
 73   */
 74   this.config = config || {};
 75 
 76   /**
 77     @name Site#title
 78     @public
 79     @type String
 80     @description The site title
 81   */
 82   this.title = this.config.title || "Untitled";
 83 
 84   // Set a private instance of app to the pages object
 85   this.pages._app = this.app;
 86 };
 87 
 88 /*
 89   @scope Site.prototype
 90 */
 91 Site.prototype = Site.fn = new (function(){
 92   /*
 93     @name Site#extend
 94     @public
 95     @function
 96     @description Extend the Site object instance
 97     @param {Object} object An object to extend the Site instance with
 98     @return The current instance of Site
 99   */
100   this.extend = function(object) {
101     return base.extend(this, object);
102   }
103 
104   /*
105     @name Site#init
106     @public
107     @function
108     @description The method used to initialize the layout or template scope with in
109     the site object and to provide bindings
110     @param {Object} scope The layout or template scope
111     @param {Object} bindings The key to value map of bindings
112     @param {Object} propagateTo A key as a value to a sub object to an array map of 
113     bindings to apply to sub objects. Ex: Site.init({partial : partial}, {partial : ['partial']});
114     @return The site instance
115   */
116   this.init = function(scope, bindings, propagateTo) {
117     var prop, i
118 
119     this.scope    = scope || {};
120     this.bindings = bindings || {};
121 
122     if (typeof propagateTo === 'object') {
123       for (prop in propagateTo) {
124         if (typeof propagateTo[prop] === 'object' && propagateTo[prop].length) {
125           for (i = propagateTo[prop].length - 1; i >= 0; i--) {
126             if (typeof this[prop].bindings !== 'object') {
127               this[prop].bindings = {};
128             }
129 
130             if (typeof this.bindings[propagateTo[prop][i]] === 'function') {
131               this[prop].bindings[propagateTo[prop][i]] = (function(func, scope) {
132                 return function() {
133                   return func.apply(scope, arguments);
134                 };
135               })(this.bindings[propagateTo[prop][i]], this.scope);
136             }
137             else {
138               this[prop].bindings[propagateTo[prop][i]] = this.bindings[propagateTo[prop][i]];
139             }
140           }
141         }
142       }
143     }
144 
145     return this;
146   };
147 
148   /**
149     @name Site#templates
150     @public
151     @type Object
152     @description The templates object
153   */
154   this.templates = {
155     /**
156       @name Site#templates#globals
157       @public
158       @type Object
159       @description The global variables for a template
160     */
161     globals : {}
162 
163     /**
164       @name Site#templates#path
165       @public
166       @type String
167       @description The default path for a template files
168     */
169   , path : 'templates/'
170     /*
171       @name Site#templates#fetch
172       @public
173       @function
174       @description Fetches a templates contents
175       @param {String} name The name of the template
176       @return The template partial
177     */
178   , fetch : function(name, data, path) {
179       return this.bindings.partial((path || this.path) + name, extend(this.globals, data || {}));
180     }
181 
182   };
183 
184   /**
185     @name Site#pages
186     @public
187     @type Object
188     @description The pages object
189   */
190   this.pages = {
191       _app      : null
192     , active    : {
193         page  : null
194       , fetch : function(){
195         return page && page.fetch? this.page.fetch() : null;
196       }
197     }
198     , _pages    : {}
199     , _helpers  : {}
200       /*
201         @name Site#pages#create
202         @public
203         @function
204         @description Creates an instance of Site.Page
205         @param {String} page The page name to create
206         @param {Object} options The options for the new Site.Page instance
207         @return The pages object
208       */
209     , create : function(name, options) {
210         var page, i, self = this
211 
212         if (typeof name === 'string' && typeof options === 'object' && options.active !== false) {
213           page = this._pages[name] = new Page(name, options, this);
214 
215           if (typeof page.routes === 'object' && page.routes.length && page.bound === false) {
216             for (i = page.routes.length - 1; i >= 0; i--) {
217               this._app.event.once(page.routes[i], function(event){
218                 page.bound = true;
219 
220                 self.active.page = page;
221                 event.respond({
222                   page : name
223                 }, {
224                   view    : page.view,
225                   layout  : (options.layout? options.layout : options.layout === false? 'empty' : false)
226                 });
227               });
228             }
229           }
230         }
231 
232         return this;
233       }
234 
235       /*
236         @name Site#pages#extend
237         @public
238         @function
239         @description Extends the Site object's pages object
240         @param {Object} pages An object to extend the Site object's pages object with
241         @return The pages object
242       */
243     , extend : function(pages){
244         var page
245 
246         for (page in pages) {
247           this.create(page, pages[page]);
248         }
249 
250         return this;
251       }
252 
253       /*
254         @name Site#pages#each
255         @public
256         @function
257         @description Enumerates over each page in the pages object
258         @param {Function} callback Callback to excute on each iteration on the object
259         @return The pages object
260       */
261     , each : function(selector, callback){
262         var page, pages
263 
264         callback = (typeof selector === 'function' ? selector : 
265                    (typeof callback === 'function' ? callback :
266                     false));
267 
268         selector = (typeof selector === 'string' ? selector : '*');
269 
270         pages = this.query(selector);
271 
272         for (page in pages) {
273           if (! pages[page] instanceof Site.Page) {
274             throw ([page, ' is not a valid instance of SinglePageApplication.Site.Page'].join(''));
275           }
276           else if (typeof callback === 'function') {
277             callback.apply(pages[page], [page, pages[page]]);
278           }
279         }
280 
281         return this;
282       }
283 
284       /*
285         @name Site#pages#get
286         @public
287         @function
288         @description Returns the instance of a page by name
289         @param {String} name The name of the page
290         @return The page instance if found, else all
291       */
292     , get  : function(name) {
293         if (typeof this._pages[name] !== 'undefined') {
294           return this._pages[name];
295         }
296 
297         return this._pages;
298       }
299 
300        /*
301         @name Site#pages#query
302         @public
303         @function
304         @description Returns the instance of a page by name
305         @param {String} query The query to filter results
306         @return An array of results
307       */
308     , query : function(query) {
309         var page, pages = this.get(), results = [], container
310 
311         if (typeof query !== 'string' && !query.length) {
312           throw "Site.pages.query() needs a valid query string!";
313         }
314 
315         for (page in pages) {
316           container = pages[page].container; // cache
317           delete pages[page].container; // remove because of circular reference
318           
319           if (base.QueryObject.engine.match(query, pages[page]).length) {
320             results.push(pages[page]);
321           }
322 
323           pages[page].container = container; // re-attache
324         }
325 
326         return results;
327       }
328 
329 
330       /*
331         @name Site#pages#fetch
332         @public
333         @function
334         @description Returns the instance of a page by name and the rendered partial
335         @param {String} name The name of the page
336         @return The page instance
337       */
338     , fetch  : function(name) {
339         var page, tpl
340         if (typeof (page = this._pages[name]) !== 'undefined') {
341           tpl = (page.template? 'templates/' + page.template : 
342                   (page.view? page.view : ''));
343 
344           page.partial = this.bindings.partial(tpl, {
345               page : name,
346               view : page.view,
347               data : page.data || {}
348           });
349 
350           return page;
351         }
352 
353         return null;
354       }
355   };
356 
357 
358 });
359 
360 Site.Page = Page;
361 exports   = Site;
