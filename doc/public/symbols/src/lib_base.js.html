 17 /*
 18   @module base

 22 var utils         = require('utilities')
 23   , jsonSelect    = require('JSONSelect')
 24 
 25 /**
 26   @name base
 27   @namespace Base namespace
 28   @version 0.5
 29 */
 30 var base = {};
 31 
 32 /**
 33   @name base#extend
 34   @public
 35   @static
 36   @function
 37   @description Extends an object
 38   @param {Object} object The main object to extend
 39   @param {, Object} objectn Any number of objects to extend the main object with
 40   @return The main object extended
 41 */
 42 base.extend = function() {
 43   var object
 44     , objectCount
 45     , i        = 0
 46     , objects  = this.makeArray(arguments);
 47   
 48   object      = objects.shift();
 49   objectCount = objects.length;
 50 
 51   for ( ; i < objectCount; i++) {
 52     object = utils.mixin(object, objects[i]);
 53   }
 54 
 55   return object;
 56 };
 57 
 58 /**
 59   @name base#makeArray
 60   @public
 61   @static
 62   @function
 63   @description Turns an object into a array. Usefule for the conversion of an arguments object
 64   to an array.
 65   @param {Object} object The object to be converted
 66   @return The new array
 67 */
 68 base.makeArray = function() { 
 69   return Array.prototype.splice.apply(arguments[0], [0]); 
 70 };
 71 
 72 /**
 73   @name base#die
 74   @public
 75   @static
 76   @function
 77   @description Exits the process and logs any number of input to the console.
 78   @param {, Mixed} input Input to be logged to the console before exiting.
 79 */
 80 base.die = function() { 
 81   console.log.error.apply(console, arguments); 
 82   process.exit(); 
 83 };

 97 
 98 /**
 99   @name base.QueryObject
100   @constructor
101   @description Constructs a new instance of a QueryObject, using JSONSelect as a default query engine.
102   @param {Object} object The object to query
103 */
104 base.QueryObject = function (object, config) {
105   /**
106     @name base.QueryObject#object
107     @public
108     @type object
109     @description The object all queries are made agaisnt.
110   */
111   this.object = object || {};
112 
113   /**
114     @name base.QueryObject#engine
115     @public
116     @type object
117     @description The engine to use to query the object
118   */
119   this.engine = config? config.engine : jsonSelect;
120 };
121 
122 
123 base.QueryObject.prototype = new function () {
124   /*
125     @scope base.QueryObject.prototype
126   */
127   /*
128     @name base.QueryObject#query
129     @public
130     @function
131     @description Queries the object for matches and returns the data using the query engine.
132     @param {String} query A query used and interpreted by the engine set, defaults to JSONSelect.
133     @return The matched results
134   */
135   this.query = function(query) {
136     return this.engine.match(query, this.object);
137   };
138 };
