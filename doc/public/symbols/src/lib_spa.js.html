 56     @name SinglePageApplication#root

 72     @name SinglePageApplication#event

100   on   = this.event.on;
101   emit = this.event.emit;
102   (function(props){
103     var i;
104 
105     for (i = 0; i < props.length; i++) {
106       this[props[i]] = (function(old){
107         return function(){
108           var args = makeArray(arguments)
109             , i
110             , ns
111             , subns
112             , event
113             , index
114 
115           // Prevent listening for everything for now
116           if (args[0] === '*') {
117             return this;
118           }
119 
120           args[0] = (args[0].length > 1? args[0].replace('/','.') : args[0]);
121           ns      = args[0].split('.')[0];
122           subns   = args[0].split('').reverse().join('').split('.')[0];
123 
124           for (i = args.length - 1; i >= 0; i--) {
125             if (typeof args[i] === 'string' && ! args[i].length) {
126               delete args[i];
127             }
128           }
129 
130           if (!!~ (index = self.routes.indexOf(args[0]))) {
131             delete self.routes[index];
132           }
133 
134           if (!~ (self.routes.indexOf(ns)) && ns.length) {
135             self.routes.push(ns);
136           }
137 
138           if (args.length && typeof args[0] === 'string' && typeof args[1] === 'function') {
139             if (!~ self.routes.indexOf(args[0])) {
140               self.routes.push(args[0]);
141             }
142 
143             old.apply(self.event, args);
144           }
145           
146           event = args.shift();
147 
148           if (subns && args.length && typeof args[0] === 'string' && typeof args[1] === 'function') {
149             old.apply(self.event, [event.replace(subns, '*')].concat(args));
150           }
151           
152           return this;
153         }
154       })(this[props[i]]);
155     }
156   }).apply(this.event, [['on', 'once']]);
157 }

159 /**

162 geddy.SinglePageApplication.prototype = geddy.SinglePageApplication.fn = new (function(){
163   "use strict";
164 
165   var init
166     , interfaces
167     , filterRouteEvents
168     , CallbackInterface
169 
170   // Private
171   filterRouteEvents = function(string, returnStd) {
172     var routeEvents   = []
173       , stdEvents     = []
174       , i
175       , event
176       , events
177 
178     events = string.split(',');
179 
180     for (i = events.length - 1; i >= 0; i--) {
181       if (!!~ (events[i].indexOf('route:'))) {
182         routeEvents.push(events[i].split(':')[1]);
183       }
184       else {
185         stdEvents.push(events[i]);
186       }
187     }
188 
189     return returnStd === true? stdEvents : routeEvents.join(',');
190   };
191 
192   init = function() {
193     var self       = this
194       , controller = this.controller 
195       , prop
196       , on
197       , once
198       , index
199 
200 
201     if (typeof controller != 'object') {
202       this.root.log.error("Invalid controller");
203     }
204 
205     for (prop in interfaces.controller) {
206       controller[prop]  = interfaces.controller[prop];
207       controller.spa    = this;
208     }
209 
210     once = self.event.once;
211     controller.on = function(){
212       var args  = makeArray(arguments)
213         , event = args[0]
214         , index
215         , i
216         , routeEvents
217 
218       if ((routeEvents = filterRouteEvents(event)).length) {
219         args.shift();
220         routeEvents = routeEvents.split(',');
221         
222         for (i = routeEvents.length - 1; i >= 0; i--) {
223           self.event.once.apply(self.event, [routeEvents[i]].concat(args));
224         }
225       }
226       else {
227         once.apply(this, args);
228       }
229 
230       return this;
231     }
232 
233     controller.view = this.view;
234 
235     controller.respondsWith = ['html', 'json'].concat(this.routing.types || []);
236 
237     this.controller = controller;
238 
239     return this;
240   }

242   /**
243     @name CallbackInterface
244     @constructor
245     @param {BaseController} controller The main controller receiving the event
246     @param {Object} data Data to be sent with the callback
247     @param {String} type The event type or route
248     @return An instance of the CallbackInterface extended by the data Object
249   */
250   CallbackInterface = function(controller, data, type) {
251     /**
252       @name CallbackInterface#type
253       @public
254       @type String
255       @description The event type or route
256     */ 
257     this.type    = type;
258 
259     /**
260       @name CallbackInterface#respond
261       @public
262       @function
263       @type Object
264       @description The respond callback that directly calls controller.respond();
265     */ 
266     this.respond = function(data, options) {
267       data    = data || {};
268       options = options || {};
269 
270       var format    = (options.format? options.forma || controller.spa.config.defaultFormat || 'html' : 'html')
271         , viewPath  = (options.path? options.path.views || controller.spa.config.path.views || 'app/views/' : 'app/views/')
272         , viewArgs  = extend(data || {}, {})
273         , options   = extend(options || {}, {
274             format    : format
275           , template  : viewPath + (options.view || 'index')
276           , layout    : viewPath + 'layouts/' + (options.layout || controller.spa.config.layout || 'empty')
277         });
278 
279       delete options.view;
280 
281       controller.app.viewHelpers = controller.spa.site.pages._helpers = extend(controller.app.viewHelpers, controller.spa.view.helpers.get());
282 
283       return controller.respond.apply(controller, [viewArgs, options]);
284     };
285 
286     return extend(this, data);
287   };

291   interfaces = {
292     /* 
293       Standard overrides index, add, edit, create, update, destroy 
294       Routing variables - (/)(/:subject)(/:control)(/:action)(/:data)(/:$1)(/:$2)(/:$3)(/:$4)(/:$5)
295     */
296     controller : extend({
297       index : function(request, response, params) {
298         var self = this
299           , eventType
300           , i
301       
302         //console.log(params.subject, params.control, params.actvitity, params.data);
303 
304         if (params.data) {
305           eventType = [params.subject, params.control, params.actvitity, params.data].join('.');
306         }
307         else if (params.actvitity) {
308           eventType = [params.subject, params.control, params.actvitity].join('.');
309         }
310         else if (params.control) {
311           eventType = [params.subject, params.control].join('.');
312         }
313         else if (params.subject && !!~ (this.spa.routes.indexOf(params.subject))) {
314           eventType = params.subject;
315         }
316         else if (!params.subject) {
317           eventType = this.spa.config.defaultSubject || '/';
318         }
319 
320         if (!!~ this.spa.routes.indexOf(eventType)) {
321           this.spa.event.emit(eventType, new CallbackInterface(this, params, eventType));
322 
323           if (!!~ this.spa.routes.indexOf(params.subject + '.*')) {
324             this.spa.event.emit(params.subject + '.*', new CallbackInterface(this, params, eventType));
325           }
326         }
327         else {
328           this.spa.event.emit('404',  new CallbackInterface(this, params, eventType));
329         }
330       }
331     }, new (function(){
332         var props = ['add', 'edit', 'create', 'update', 'destroy'];
333 
334         while(props.length)
335           this[props.shift()] = noop;
336 
337         return this;
338     }))
339   };

344   /*
345     @name SinglePageApplication#registerController
346     @public
347     @function
348     @description Registers the controller for the SPA
349     @param {BaseController} controller An instance of controller.BaseController provided
350     by the Geddy Web Framework.
351     @param {Object} routing Rating configuration and route definitions
352     @return The current instance of SinglePageApplication
353   */
354   this.registerController = function(controller, routing) {
355     this.controller = controller || !function(){ throw "A SinglePageApplication needs a controller"; die(); }();
356     this.routes     = !routing? [] : routing.routes || [];
357     this.routing    = routing || {};
358 
359     delete this.routing.routes;
360 
361     return init.call(this);
362   };

382   this.view = {
383     /*
384       @name SinglePageApplication#view#helpers
385       @public
386       @object
387       @description View helpers
388     */
389     helpers : {
390         _helpers    : {}
391         /*
392           @name SinglePageApplication#view#helpers#add
393           @public
394           @function
395           @description Adds a view helper for the view
396           @param {String} alias An alias to the view helper
397           @param {Function|String} value The view helper value
398           @return The view helper object
399         */
400       , add : function(alias, value) {
401           var prop;
402 
403           if (typeof alias === 'string' && typeof value !== 'undefined') {
404             this._helpers[alias] = value;
405           }
406           else if (typeof alias === 'object') {
407             for (prop in alias) {
408               this.add(prop, alias[prop]);
409             }
410           }
411 
412           return this;
413         }
414 
415         /*
416           @name SinglePageApplication#view#helpers#remove
417           @public
418           @function
419           @description Removes a view helper for the view
420           @param {String} alias An alias to the view helper value
421           @return The view helper object
422         */
423       , remove : function(alias) {
424           if (typeof alias === 'string') {
425             delete this._helpers[alias];
426           }
427 
428           return this;
429         }
430 
431        /*
432           @name SinglePageApplication#view#helpers#get
433           @public
434           @function
435           @description Gets a view helpers value
436           @param {String} alias An alias to the view helper value
437           @return The view helper if found, else the whole object
438         */
439       , get : function(alias) {
440           if (typeof alias === 'string' && this._helpers[alias]) {
441             return this._helpers[alias];
442           }
443 
444           return this._helpers;
445         }
446     }
447   };
